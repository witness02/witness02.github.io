如何在Golang中调用汇编

简单的例子

  1. 在GOPATH下的add目录下创建add.go和add_amd64.s文件（不同的平台使用不同的汇编文件及命名）
    a. add.go

    package add
func AddTwo(a, b uint64) uint64
    
func Add(a, b, c uint64) uint64

    b. add_amd64.s

    TEXT    ·AddTwo+0(SB),$0-24
    MOVQ    a+0(FP),BX
    
    MOVQ    b+8(FP),BP
    
    ADDQ    BP,BX
    
    MOVQ    BX,res+16(FP)
    
    RET     
    
TEXT    ·Add+0(SB),$0-32
    
    MOVQ    a+0(FP),BX
    
    MOVQ    b+8(FP),BP
    
    ADDQ    BP,BX
    
    MOVQ    c+16(FP),BP
    
    ADDQ    BP,BX
    
    MOVQ    BX,res+24(FP)
    
    RET

  2. 在任意目录创建main.go

package main

import (
    "fmt"
    "add"
)

func main() {
    fmt.Println(add.Add(1, 2, 15))
    fmt.Println(add.AddTwo(1, 2))
}
  1. 执行go run main.go
